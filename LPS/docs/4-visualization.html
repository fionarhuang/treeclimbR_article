is_seq <- function(x, by, num_seq, tol = 1e-5){
  x1 <- c(FALSE, abs(diff(x) - by) > tol)
  x2 <- cumsum(x1)+1
  x3 <- table(x2)
  x4 <- which(x3 >= num_seq)
  fx <- x[x2 %in% x4]
  return(fx)
}
